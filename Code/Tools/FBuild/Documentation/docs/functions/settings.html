Settings

Defines global FASTBuild settings.

Settings { // General .Environment // (optional) Array of environment variables to use // Caching .CachePath // (optional) Path to cache location .CachePluginDLL // (optional) User plugin to manage cache back-end // Distribution .Workers // (optional) Fixed list of workers if not using automatic discovery .WorkerConnectionLimit // (optional) Limit number of connected workers, default: 15 .DistributableJobMemoryLimitMiB // (optional) Limit memory used locally to prep jobs // default: 2048 (1024 for X86) .WorkerListRefreshLimit // (optional) When not using a static worker list, // max number of seconds to refresh the network share worker list. // default: 300 (5 minutes, in case workers with specific tags are rebooting) // 0 means no retries // -1 means infinite retries .WorkerConnectionRetryLimit // (optional) When a worker does not respond to a connection request, // max number of seconds to retry a worker connection. // default: 300 (5 minutes, in case workers with specific tags are rebooting) // 0 means no retries // -1 means infinite retries // Sandbox .SandboxEnabled // (optional) Enable local worker execution sandbox, default: false .SandboxExe // (optional) Local sandbox executable, default: empty string // Example Windows sandbox tool: // https://github.com/iceb0y/windows-container .SandboxArgs // (optional) Local sandbox args, default: empty string .SandboxTmp // (optional) Local sandbox tmp dir, default: 'sandbox' // For now, FBuild only sets low integrity // permissions for the dir on Windows systems. // Tags .LocalWorkerTags // (optional) List of local worker tags // default: empty list // example: // { // 'TopDownMemory', // 'OS=Win-7-64' // } }
